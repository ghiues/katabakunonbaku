        let questions = [];
        let currentIdx = 0;
        let score = 0;
        let totalQuestions = 0;

        // Fungsi acak array
        function shuffle(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
        }

        function startQuiz(limit) {
            // Memproses rawData menjadi array objek
            questions = rawData.map(item => {
                let parts = item.split("|");
                return { baku: parts[0], tidakBaku: parts[1] };
            });

            shuffle(questions); // Acak seluruh bank soal

            // Batasi jumlah soal sesuai pilihan user
            if(limit < questions.length) {
                questions = questions.slice(0, limit);
            }
            
            totalQuestions = questions.length;
            currentIdx = 0;
            score = 0;

            document.getElementById('setup-screen').style.display = 'none';
            document.getElementById('quiz-screen').style.display = 'block';

            loadQuestion();
        }

        function loadQuestion() {
            document.getElementById('btn-next').style.display = 'none';
            document.getElementById('feedback').innerText = '';
            
            document.getElementById('q-counter').innerText = `Soal ${currentIdx + 1} dari ${totalQuestions}`;
            document.getElementById('score-counter').innerText = `Benar: ${score}`;

            const qData = questions[currentIdx];
            
            // Format jawaban agar huruf awalnya besar (Capitalize)
            const cap = (str) => str.charAt(0).toUpperCase() + str.slice(1);

            let options = [
                { text: cap(qData.baku), isCorrect: true },
                { text: cap(qData.tidakBaku), isCorrect: false }
            ];
            shuffle(options);

            const optionsArea = document.getElementById('options-area');
            optionsArea.innerHTML = '';

            options.forEach(opt => {
                let btn = document.createElement('button');
                btn.className = 'btn-option';
                btn.innerText = opt.text;
                btn.onclick = () => checkAnswer(btn, opt.isCorrect, cap(qData.baku));
                optionsArea.appendChild(btn);
            });
        }

        function checkAnswer(selectedBtn, isCorrect, correctWord) {
            const buttons = document.querySelectorAll('.btn-option');
            buttons.forEach(btn => btn.disabled = true);

            const feedback = document.getElementById('feedback');

            if (isCorrect) {
                selectedBtn.classList.add('correct');
                feedback.innerText = "✅ Tepat Sekali!";
                feedback.style.color = "var(--correct)";
                score++;
                document.getElementById('score-counter').innerText = `Benar: ${score}`;
            } else {
                selectedBtn.classList.add('wrong');
                feedback.innerText = `❌ Kurang Tepat! Yang baku adalah: ${correctWord}`;
                feedback.style.color = "var(--wrong)";
                
                // Cari dan warnai tombol yang benar
                buttons.forEach(btn => {
                    if (btn.innerText === correctWord) {
                        btn.classList.add('correct');
                    }
                });
            }

            document.getElementById('btn-next').style.display = 'block';
        }

        function nextQuestion() {
            currentIdx++;
            if (currentIdx < totalQuestions) {
                loadQuestion();
            } else {
                showResult();
            }
        }

        function showResult() {
            document.getElementById('quiz-screen').style.display = 'none';
            document.getElementById('result-screen').style.display = 'block';

            let finalScore = Math.round((score / totalQuestions) * 100);
            document.getElementById('final-score').innerText = finalScore;

            let msg = document.getElementById('final-message');
            if (finalScore >= 90) msg.innerText = "Luar Biasa! Kemampuan Kosakata Anda setingkat Editor KBBI 🏆";
            else if (finalScore >= 70) msg.innerText = "Sangat Bagus! Anda sudah sangat siap untuk UTBK PPU/PBM 👍";
            else if (finalScore >= 50) msg.innerText = "Lumayan. Terus asah insting bahasa Indonesia Anda! 📖";
            else msg.innerText = "Jangan menyerah! Yuk hafalkan lagi daftar kata baku perlahan-lahan 💪";
